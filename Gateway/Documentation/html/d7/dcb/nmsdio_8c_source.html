121 static int sdio_write(uint32_t addr, uint8_t *buf, uint32_t size)

123  tstrNmSdioCmd53 cmd;
124  int nblk, nleft;
125  cmd.read_write = 1;
126  if (addr > 0) {

129  */
130  if (size & 0x3) {
131  size += 4;
132  size &= ~0x3;
133  }
134 
135  /*
136  * func 0 access
137  */
138  cmd.function = 0;
139  cmd.address = 0x10f;
140  } else {
141  /*
142  * has to be word aligned...
143  */
144  if (size & 0x3) {
145  size += 4;
146  size &= ~0x3;
147  }
148 
149  /*
150  * func 1 access
151  */
152  cmd.function = 1;
153  cmd.address = 0;
154  }
155 
156  nblk = size / WILC_SDIO_BLOCK_SIZE;
157  nleft = size % WILC_SDIO_BLOCK_SIZE;
158 
159  if (nblk > 0) {

161  cmd.increment = 1;
162  cmd.count = nblk;
163  cmd.buffer = buf;

172  }
173  if (addr > 0)

177  }

433 static int sdio_int_enable(void)
434 {
435  uint32_t reg;
436  int ret;
440  ret = sdio_read_reg(NMI_PIN_MUX_0, &reg);
441  if (ret) {
442  M2M_ERR("[nmi sdio]: Failed read reg (%08x)...\r\n", NMI_PIN_MUX_0);
443  return M2M_ERR_INIT;
444  }
445  reg |= (1<<8);
446  ret = sdio_write_reg(NMI_PIN_MUX_0, reg);
447  if (ret) {
448  M2M_ERR("[nmi sdio]: Failed write reg (%08x)...\r\n", NMI_PIN_MUX_0);
449  return M2M_ERR_INIT;
450  }
451 
455  ret = sdio_read_reg(NMI_INTR_ENABLE, &reg);
456  if (ret) {
457  M2M_ERR("[nmi sdio]: Failed read reg (%08x)...\r\n", NMI_INTR_ENABLE);
458  return M2M_ERR_INIT;
459  }
460  reg |= (1 << (27));
461  #ifdef INT_BASED_TX
462  reg |= (1 << (28));
463  #endif
464  ret = sdio_write_reg(NMI_INTR_ENABLE, reg);
465  if (ret) {
466  M2M_ERR("[nmi sdio]: Failed write reg (%08x)...\r\n", NMI_INTR_ENABLE);
467  return M2M_ERR_INIT;
468  }
469  return M2M_SUCCESS;
470 }

472 /*
473 * @fn nm_sdio_init
474 * @brief Initialize the SDIO
475 * @return M2M_SUCCESS in case of success and M2M_ERR_BUS_FAIL in case of failure
476 * @author A. Abozaeid
477 * @date 13 September 2015
478 * @version 1.0
479 */
480 sint8 nm_sdio_init(void)
481 {
482  tstrNmSdioCmd52 cmd;
483  int loop;
484 
485  /*
486  * function 0 csa enable
487  */
488  cmd.read_write = 1;
489  cmd.function = 0;
490  cmd.raw = 1;
491  cmd.address = 0x100;
492  cmd.data = 0x80;
493  if (nm_bus_ioctl(NM_BUS_IOCTL_CMD_52, &cmd)) {
494  M2M_ERR("Fail cmd 52, enable csa\r\n");
495  return M2M_ERR_BUS_FAIL;
496  }
497 
498  /*
499  * function 0 block size
500  */
501  if (sdio_set_func0_block_size(WILC_SDIO_BLOCK_SIZE)) {
502  M2M_ERR("Fail cmd 52, set func 0 block size\r\n");
503  return M2M_ERR_BUS_FAIL;
504  }
505 
506  /*
507  * enable func1 IO
508  */
509  cmd.read_write = 1;
510  cmd.function = 0;
511  cmd.raw = 1;
512  cmd.address = 0x2;
513  cmd.data = 0x2;
514  if (nm_bus_ioctl(NM_BUS_IOCTL_CMD_52, &cmd)) {
515  M2M_ERR("Fail cmd 52, set IOE register\r\n");
516  return M2M_ERR_BUS_FAIL;
517  }
518 
519  /*
520  * make sure func 1 is up
521  */
522  cmd.read_write = 0;
523  cmd.function = 0;
524  cmd.raw = 0;
525  cmd.address = 0x3;
526  loop = 3;
527  do {
528  cmd.data = 0;
529  if (nm_bus_ioctl(NM_BUS_IOCTL_CMD_52, &cmd)) {
530  M2M_ERR("Fail cmd 52, get IOR register\r\n");
531  return M2M_ERR_INIT;
532  }
533  if (cmd.data == 0x2)
534  break;
535  } while (loop--);
536 
537  if (loop <= 0) {
538  M2M_ERR("Fail func 1 is not ready\r\n");
539  return M2M_ERR_INIT;
540  }
541 
542  /*
543  * func 1 is ready, set func 1 block size
544  */
545  if (sdio_set_func1_block_size(WILC_SDIO_BLOCK_SIZE)) {
546  M2M_ERR("Fail set func 1 block size\r\n");
547  return M2M_ERR_BUS_FAIL;
548  }
549 
550  /*
551  * func 1 interrupt enable
552  */
553  cmd.read_write = 1;
554  cmd.function = 0;
555  cmd.raw = 1;
556  cmd.address = 0x4;
557  cmd.data = 0x3;
558  if (nm_bus_ioctl(NM_BUS_IOCTL_CMD_52, &cmd)) {
559  M2M_ERR("Fail cmd 52, set IEN register\r\n");
560  return M2M_ERR_BUS_FAIL;
561  }
562 
563  sdio_int_enable();
564 
565  return M2M_SUCCESS;
566 }
