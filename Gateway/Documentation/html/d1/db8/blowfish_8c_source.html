290 error_t blowfishExpandKey(BlowfishContext *context, const uint8_t *salt,
291  size_t saltLen, const uint8_t *key, size_t keyLen)
292 {
293  uint_t i;
294  uint_t j;
295  uint32_t value;
296  size_t keyIndex;
297  size_t saltIndex;
298  uint8_t data[BLOWFISH_BLOCK_SIZE];
299 
300  //Check the length of the salt
301  if(saltLen != 0 && saltLen < BLOWFISH_BLOCK_SIZE)
302  return ERROR_INVALID_PARAMETER;
303 
304  //Check the length of the key
305  if(keyLen == 0)
306  return ERROR_INVALID_KEY_LENGTH;
307 
308  //Initialize variables
309  keyIndex = 0;
310  saltIndex = 0;
311 
312  //In the first phase of setting up the P-array, the secret key K is
313  //cyclically XOR'd into the P-array
314  for(i = 0; i < 18; i++)
315  {
316  //This is done by taking successive 32-bit quantities of K
317  for(value = 0, j = 0; j < 4; j++)
318  {
319  //Read current byte
320  value = (value << 8) | key[keyIndex];
321 
322  //Wrap around to the start if the end of the key is reached
323  if(++keyIndex >= keyLen)
324  {
325  keyIndex = 0;
326  }
327  }
328 
329  //XOR the 32-bit quantities of K with the P-array elements
330  context->p[i] ^= value;
331  }
332 
333  //Initialize the 64-bit data block to all zeroes
334  cryptoMemset(data, 0, BLOWFISH_BLOCK_SIZE);
335 
336  //In the second phase, the 64-bit data block is repeatedly encrypted to
337  //yield the final P-array
338  for(i = 0; i < 18; i += 2)
339  {
340  //The data block is XOR'd with the salt
341  blowfishXorBlock(data, salt, saltLen, &saltIndex);
342 
343  //Encrypt the data block
344  blowfishEncryptBlock(context, data, data);
345  context->p[i] = LOAD32BE(data);
346  context->p[i + 1] = LOAD32BE(data + 4);
347  }
348 
349  //The process used for the second phase of the P-array initialisation is
350  //now repeated to initialize the S-boxes
351  for(i = 0; i < 256; i += 2)
352  {
353  //The data block is XOR'd with the salt
354  blowfishXorBlock(data, salt, saltLen, &saltIndex);
355 
356  //Encrypt the data block
357  blowfishEncryptBlock(context, data, data);
358  context->s1[i] = LOAD32BE(data);
359  context->s1[i + 1] = LOAD32BE(data + 4);
360  }
361 
362  //Initialize S-box 2
363  for(i = 0; i < 256; i += 2)
364  {
365  //The data block is XOR'd with the salt
366  blowfishXorBlock(data, salt, saltLen, &saltIndex);
367 
368  //Encrypt the data block
369  blowfishEncryptBlock(context, data, data);
370  context->s2[i] = LOAD32BE(data);
371  context->s2[i + 1] = LOAD32BE(data + 4);
372  }
373 
374  //Initialize S-box 3
375  for(i = 0; i < 256; i += 2)
376  {
377  //The data block is XOR'd with the salt
378  blowfishXorBlock(data, salt, saltLen, &saltIndex);
379 
380  //Encrypt the data block
381  blowfishEncryptBlock(context, data, data);
382  context->s3[i] = LOAD32BE(data);
383  context->s3[i + 1] = LOAD32BE(data + 4);
384  }
385 
386  //Initialize S-box 4
387  for(i = 0; i < 256; i += 2)
388  {
389  //The data block is XOR'd with the salt
390  blowfishXorBlock(data, salt, saltLen, &saltIndex);
391 
392  //Encrypt the data block
393  blowfishEncryptBlock(context, data, data);

410 void blowfishEncryptBlock(BlowfishContext *context, const uint8_t *input,
411  uint8_t *output)
412 {
413  uint_t i;
414  uint32_t left;
415  uint32_t right;
416  uint32_t temp;
417 
418  //Divide the plaintext into two 32-bit halves (L and R)
419  left = LOAD32BE(input + 0);
420  right = LOAD32BE(input + 4);
421 
422  //16 rounds of computation are needed
423  for(i = 0; i < 16; i += 2)
424  {
425  //Apply odd round function
426  left ^= context->p[i];
427  temp = context->s1[(left >> 24) & 0xFF];
428  temp += context->s2[(left >> 16) & 0xFF];
429  temp ^= context->s3[(left >> 8) & 0xFF];
430  temp += context->s4[left & 0xFF];
431  right ^= temp;
432 
433  //Apply even round function
434  right ^= context->p[i + 1];
435  temp = context->s1[(right >> 24) & 0xFF];
436  temp += context->s2[(right >> 16) & 0xFF];
437  temp ^= context->s3[(right >> 8) & 0xFF];
438  temp += context->s4[right & 0xFF];
439  left ^= temp;
440  }
441 
442  //The rounds are followed by two final XOR's with the last elements of
443  //the P-array
444  left ^= context->p[16];
445  right ^= context->p[17];
446 
447  //Recombine L and R
448  STORE32BE(right, output + 0);
449  STORE32BE(left, output + 4);
450 }

460 void blowfishDecryptBlock(BlowfishContext *context, const uint8_t *input,
461  uint8_t *output)
462 {
463  uint_t i;
464  uint32_t left;
465  uint32_t right;
466  uint32_t temp;
467 
468  //Divide the ciphertext into two 32-bit halves (L and R)
469  right = LOAD32BE(input + 0);
470  left = LOAD32BE(input + 4);
471 
472  //16 rounds of computation are needed
473  for(i = 16; i > 0; i -= 2)
474  {
475  //Apply even round function
476  right ^= context->p[i + 1];
477  temp = context->s1[(right >> 24) & 0xFF];
478  temp += context->s2[(right >> 16) & 0xFF];
479  temp ^= context->s3[(right >> 8) & 0xFF];
480  temp += context->s4[right & 0xFF];
481  left ^= temp;
482 
483  //Apply odd round function
484  left ^= context->p[i];
485  temp = context->s1[(left >> 24) & 0xFF];
486  temp += context->s2[(left >> 16) & 0xFF];
487  temp ^= context->s3[(left >> 8) & 0xFF];
488  temp += context->s4[left & 0xFF];

490  }
491 
492  //The rounds are followed by two final XOR's with the first elements of
493  //the P-array
494  right ^= context->p[1];
495  left ^= context->p[0];
496 
497  //Recombine L and R
498  STORE32BE(left, output + 0);
499  STORE32BE(right, output + 4);
