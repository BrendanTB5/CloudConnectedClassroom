269 void systemInit(void)
270 {
271  //Execute system unlock sequence
272  SYSKEY = 0xAA996655;
273  SYSKEY = 0x556699AA;
274 
275  //Check PBDIVRDY bit
276  while (!(PB2DIV & _PB2DIV_PBDIVRDY_MASK));
277  //Configure PBCLK2 clock divisor (SYSCLK / 5);
278  PB2DIV = _PB2DIV_ON_MASK | 4;
279 
280  //Check PBDIVRDY bit
281  while (!(PB3DIV & _PB3DIV_PBDIVRDY_MASK));
282  //Configure PBCLK3 clock divisor (SYSCLK / 5);
283  PB3DIV = _PB3DIV_ON_MASK | 4;
284 
285  //Check PBDIVRDY bit
286  while (!(PB4DIV & _PB4DIV_PBDIVRDY_MASK));
287  //Configure PBCLK4 clock divisor (SYSCLK / 1);
288  PB4DIV = _PB4DIV_ON_MASK | 0;
289 
290  //Check PBDIVRDY bit
291  while (!(PB5DIV & _PB5DIV_PBDIVRDY_MASK));
292  //Configure PBCLK5 clock divisor (SYSCLK / 2);
293  PB5DIV = _PB5DIV_ON_MASK | 1;
294 
295  //Relock the SYSKEY
296  SYSKEY = 0;
297 
298  //Disable interrupts
299  __builtin_disable_interrupts();
300 
301  //Set IV
302  _CP0_BIS_CAUSE(_CP0_CAUSE_IV_MASK);
303  //Enable multi-vectored mode
304  INTCONSET = _INTCON_MVEC_MASK;
305 }
