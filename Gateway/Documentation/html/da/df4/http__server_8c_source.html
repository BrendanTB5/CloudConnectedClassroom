544 error_t httpReadStream(HttpConnection *connection,
545  void *data, size_t size, size_t *received, uint_t flags)
546 {
547  error_t error;
548  size_t n;
549 
550  //No data has been read yet
551  *received = 0;
552 
553  //Chunked encoding transfer is used?
554  if(connection->request.chunkedEncoding)
555  {
556  //Point to the output buffer
557  char_t *p = data;
558 
559  //Read as much data as possible
560  while(*received < size)
561  {
562  //End of HTTP request body?
563  if(connection->request.lastChunk)
564  return ERROR_END_OF_STREAM;
565 
566  //Acquire a new chunk when the current chunk
567  //has been completely consumed
568  if(connection->request.byteCount == 0)
569  {
570  //The size of each chunk is sent right before the chunk itself
571  error = httpReadChunkSize(connection);
572  //Failed to decode the chunk-size field?
573  if(error)
574  return error;
575 
576  //Any chunk whose size is zero terminates the data transfer
577  if(!connection->request.byteCount)
578  {
579  //The user must be satisfied with data already on hand
580  return (*received > 0) ? NO_ERROR : ERROR_END_OF_STREAM;
581  }
582  }
583 
584  //Limit the number of bytes to read at a time
585  n = MIN(size - *received, connection->request.byteCount);
586 
587  //Read data
588  error = httpReceive(connection, p, n, &n, flags);
589  //Any error to report?
590  if(error)
591  return error;
592 
593  //Total number of data that have been read
594  *received += n;
595  //Number of bytes left to process in the current chunk
596  connection->request.byteCount -= n;
597 
598  //The HTTP_FLAG_BREAK_CHAR flag causes the function to stop reading
599  //data as soon as the specified break character is encountered
600  if(flags & HTTP_FLAG_BREAK_CRLF)
601  {
602  //Check whether a break character has been received
603  if(p[n - 1] == LSB(flags))
604  break;
605  }
606  //The HTTP_FLAG_WAIT_ALL flag causes the function to return
607  //only when the requested number of bytes have been read
608  else if(!(flags & HTTP_FLAG_WAIT_ALL))
609  {
610  break;
611  }
612 
613  //Advance data pointer
614  p += n;
615  }
616  }
617  //Default encoding?
618  else
619  {
620  //Return immediately if the end of the request body has been reached
621  if(!connection->request.byteCount)
622  return ERROR_END_OF_STREAM;
623 
624  //Limit the number of bytes to read
625  n = MIN(size, connection->request.byteCount);
626 
627  //Read data
628  error = httpReceive(connection, data, n, received, flags);
629  //Any error to report?
630  if(error)
631  return error;
632 
633  //Decrement the count of remaining bytes to read
634  connection->request.byteCount -= *received;
635  }
636 
637  //Successful read operation
638  return NO_ERROR;
639 }
