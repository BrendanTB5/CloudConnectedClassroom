63 static sint8 spi_rw(uint8 *pu8Mosi, uint8 *pu8Miso, uint16 u16Sz)
64 {
65  uint16_t i;
66  uint8_t data;
67 
68  //Pull the CS pin low
69 #if defined(USE_HAL_DRIVER)
70  HAL_GPIO_WritePin(CONF_WILC_CS_GPIO, CONF_WILC_CS_PIN, GPIO_PIN_RESET);
71 #elif defined(USE_STDPERIPH_DRIVER)
72  GPIO_ResetBits(CONF_WILC_CS_GPIO, CONF_WILC_CS_PIN);
73 #endif
74 
75  //Perform SPI transfer
76  for(i = 0; i < u16Sz; i++)
77  {
78  //Full-duplex transfer?
79  if(pu8Mosi != NULL)
80  data = pu8Mosi[i];
81  else
82  data = 0x00;
83 
84  //Ensure the TX buffer is empty
85  while(!(CONF_WILC_SPI->SR & SPI_SR_TXE));
86  //Write character
87  CONF_WILC_SPI->DR = data;
88  //Wait for the operation to complete
89  while(!(CONF_WILC_SPI->SR & SPI_SR_RXNE));
90  //Read character
91  data = CONF_WILC_SPI->DR;
92 
93  //Save the received character
94  if(pu8Miso != NULL)
95  pu8Miso[i] = data;
96  }
97 
98  //Terminate the operation by raising the CS pin
99 #if defined(USE_HAL_DRIVER)
100  HAL_GPIO_WritePin(CONF_WILC_CS_GPIO, CONF_WILC_CS_PIN, GPIO_PIN_SET);
101 #elif defined(USE_STDPERIPH_DRIVER)
102  GPIO_SetBits(CONF_WILC_CS_GPIO, CONF_WILC_CS_PIN);
103 #endif
104 
105  //Successful operation
106  return M2M_SUCCESS;
107 }
