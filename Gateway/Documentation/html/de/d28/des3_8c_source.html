71 {

76  //Check key length
77  if(keyLen == 8)
78  {
79  //This option provides backward compatibility with DES, because the
80  //first and second DES operations cancel out
81  desInit(&context->k1, key, 8);
82  desInit(&context->k2, key, 8);
83  desInit(&context->k3, key, 8);
84  }
85  else if(keyLen == 16)
