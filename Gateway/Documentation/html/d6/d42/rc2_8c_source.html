109 error_t rc2InitEx(Rc2Context *context, const uint8_t *key, size_t keyLen,
110  uint_t effectiveKeyLen)
111 {
112  uint_t i;
113  uint_t t8;
114  uint8_t tm;
115 
116  //Check parameters
117  if(context == NULL || key == NULL)
118  return ERROR_INVALID_PARAMETER;
119 
120  //Make sure the key length is acceptable
121  if(keyLen < 1 || keyLen > 128)
122  return ERROR_INVALID_KEY_LENGTH;
123 
124  //Make sure the maximum effective key length is acceptable
125  if(effectiveKeyLen < 1 || effectiveKeyLen > 1024)
126  return ERROR_INVALID_KEY_LENGTH;
127 
128  //The key expansion algorithm begins by placing the supplied T-byte key
129  //into bytes L[0], ..., L[T-1] of the key buffer
130  cryptoMemcpy(context->l, key, keyLen);
131 
132  //The key expansion algorithm then computes the effective key length in
133  //bytes T8
134  t8 = (effectiveKeyLen + 7) / 8;
135 
136  //The mask TM has its 8 - (8*T8 - T1) least significant bits set
137  tm = 0xFF >> (8 * t8 - effectiveKeyLen);
138 
139  //First loop of the key expansion operation
140  for(i = keyLen; i < 128; i++)
141  {
142  context->l[i] = piTable[(context->l[i - 1] + context->l[i - keyLen]) & 0xFF];
143  }
144 
145  //The intermediate step's bitwise AND operation reduces the search space
146  //for L[128-T8] so that the effective number of key bits is T1
147  context->l[128 - t8] = piTable[context->l[128 - t8] & tm];
148 
149  //Second loop of the key expansion operation
150  for(i = 128 - t8; i > 0; i--)
151  {
152  context->l[i - 1] = piTable[context->l[i] ^ context->l[i + t8 - 1]];
153  }
154 
155  //The low-order byte of each K word is given before the high-order byte
156  for(i = 0; i < 64; i++)
157  {
158  context->k[i] = letoh16(context->k[i]);
159  }
160 
161  //No error to report
162  return NO_ERROR;
163 }
