201 error_t presentInit(PresentContext *context, const uint8_t *key, size_t keyLen)
202 {
203  uint_t i;
204  uint64_t t;
205  uint64_t kl;
206  uint64_t kh;
207 
208  //Check parameters
209  if(context == NULL || key == NULL)
210  return ERROR_INVALID_PARAMETER;
211 
212  //Check key length
213  if(keyLen != 10 && keyLen != 16)
214  return ERROR_INVALID_KEY_LENGTH;
215 
216  //PRESENT can take keys of either 80 or 128 bits
217  if(keyLen == 10)
218  {
219  //Copy the 80-bit key
220  kh = LOAD16BE(key);
221  kl = LOAD64BE(key + 2);
222 
223  //Save the 64 leftmost bits of K
224  context->ks[0] = (kh << 48) | (kl >> 16);
225 
226  //Generate round keys
227  for(i = 1; i <= 31; i++)
228  {
229  //The key register is rotated by 61 bit positions to the left
230  t = kh & 0xFFFF;
231  kh = (kl >> 3) & 0xFFFF;
232  kl = (kl << 61) | (t << 45) | (kl >> 19);
233 
234  //The left-most four bits are passed through the S-box
235  t = sbox[(kh >> 12) & 0x0F];
236  kh = (kh & 0x0FFF) | (t << 12);
237 
238  //The round counter value i is XOR-ed with bits 19, 18, 17, 16, 15 of K
239  kl ^= (uint64_t) i << 15;
240 
241  //Save the 64 leftmost bits of K
242  context->ks[i] = (kh << 48) | (kl >> 16);
243  }
244  }
245  else
246  {
247  //Copy the 128-bit key
248  kh = LOAD64BE(key);
249  kl = LOAD64BE(key + 8);
250 
251  //Save the 64 leftmost bits of K
252  context->ks[0] = kh;
253 
254  //Generate round keys
255  for(i = 1; i <= 31; i++)
256  {
257  //The key register is rotated by 61 bit positions to the left
258  t = kh;
259  kh = (t << 61) | (kl >> 3);
260  kl = (kl << 61) | (t >> 3);
261 
262  //The left-most eight bits are passed through two S-boxes
263  t = sbox[(kh >> 56) & 0x0F];
264  kh = (kh & 0xF0FFFFFFFFFFFFFF) | (t << 56);
265  t = sbox[(kh >> 60) & 0x0F];
266  kh = (kh & 0x0FFFFFFFFFFFFFFF) | (t << 60);
267 
268  //The round counter value i is XOR-ed with bits 66, 65, 64, 63, 62 of K
269  kh ^= (uint64_t) i >> 2;
270  kl ^= (uint64_t) i << 62;
271 
272  //Save the 64 leftmost bits of K
273  context->ks[i] = kh;
274  }
275  }
