234 void whirlpoolUpdate(WhirlpoolContext *context, const void *data, size_t length)
235 {
236  size_t n;
237

239  while(length > 0)

243 
244  //Copy the data to the buffer
245  cryptoMemcpy(context->buffer + context->size, data, n);
246 
247  //Update the Whirlpool context
248  context->size += n;
