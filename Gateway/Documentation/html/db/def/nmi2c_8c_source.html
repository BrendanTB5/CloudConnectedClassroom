56  sint8 nm_i2c_read_reg_with_ret(uint32 u32Addr, uint32* pu32RetVal)
57 {
58  uint8 b[6];
59  uint8 rsz;
60  tstrNmI2cDefault strI2c;
61  sint8 s8Ret = M2M_SUCCESS;
62 
63  if(u32Addr < 0xff) { /* clockless i2c */
64  b[0] = 0x09;
65  b[1] = (uint8)(u32Addr);
66  rsz = 1;
67  strI2c.u16Sz = 2;
68  } else {
69  b[0] = 0x80;
70  b[1] = (uint8)(u32Addr >> 24);
71  b[2] = (uint8)(u32Addr >> 16);
72  b[3] = (uint8)(u32Addr >> 8);
73  b[4] = (uint8)(u32Addr);
74  b[5] = 0x04;
75  rsz = 4;
76  strI2c.u16Sz = 6;
77  }
78 
79  strI2c.pu8Buf = b;
80 
81  if(M2M_SUCCESS == nm_bus_ioctl(NM_BUS_IOCTL_W, &strI2c))
82  {
83  strI2c.u16Sz = rsz;
84  if(M2M_SUCCESS != nm_bus_ioctl(NM_BUS_IOCTL_R, &strI2c))
85  {
86  //M2M_ERR("read error\r\n");
87  s8Ret = M2M_ERR_BUS_FAIL;
88  }
89  }
90  else
91  {
92  M2M_ERR("failed to send cfg bytes\r\n");
93  s8Ret = M2M_ERR_BUS_FAIL;
94  }
95 
96  if (rsz == 1) {
97  *pu32RetVal = b[0];
98  } else {
99  *pu32RetVal = b[0] | ((uint32)b[1] << 8) | ((uint32)b[2] << 16) | ((uint32)b[3] << 24);
100  }
101  return s8Ret;
102 }
