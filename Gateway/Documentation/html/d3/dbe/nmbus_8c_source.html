199 /*
200 * @fn nm_read_block
201 * @brief Read block of data
202 * @param [in] u32Addr
203 * Start address
204 * @param [out] puBuf
205 * Pointer to a buffer used to return the read data
206 * @param [in] u32Sz
207 * Number of bytes to read. The buffer size must be >= u32Sz
208 * @return M2M_SUCCESS in case of success and M2M_ERR_BUS_FAIL in case of failure
209 * @author M. Abdelmawla
210 * @date 11 July 2012
211 * @version 1.0
212 */
213 sint8 nm_read_block(uint32 u32Addr, uint8 *puBuf, uint32 u32Sz)
214 {
215  uint16 u16MaxTrxSz = egstrNmBusCapabilities.u16MaxTrxSz - MAX_TRX_CFG_SZ;
216  uint32 off = 0;
217  sint8 s8Ret = M2M_SUCCESS;
218 
219  for(;;)
220  {
221  if(u32Sz <= u16MaxTrxSz)
222  {
223  s8Ret += p_nm_read_block(u32Addr, &puBuf[off], (uint16)u32Sz);
224  break;
225  }
226  else
227  {
228  s8Ret += p_nm_read_block(u32Addr, &puBuf[off], u16MaxTrxSz);
229  if(M2M_SUCCESS != s8Ret) break;
230  u32Sz -= u16MaxTrxSz;
231  off += u16MaxTrxSz;
232  u32Addr += u16MaxTrxSz;
233  }
234  }
235 
236  return s8Ret;
237 }
