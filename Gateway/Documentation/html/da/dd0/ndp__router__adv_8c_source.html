321 
322  //Get current time
323  time = osGetSystemTime();
324 
325  //Make sure that the link is up and the service is running
326  if(interface->linkState && context->running)

408 void ndpProcessRouterSol(NetInterface *interface, Ipv6PseudoHeader *pseudoHeader,
409  const NetBuffer *buffer, size_t offset, uint8_t hopLimit)
410 {
411  error_t error;
412  uint_t n;
413  size_t length;
414  systime_t time;
415  systime_t delay;
416  NdpRouterAdvContext *context;
417  NdpRouterSolMessage *message;
418  NdpLinkLayerAddrOption *option;
419  NdpNeighborCacheEntry *entry;
420 
421  //Point to the RA service context
422  context = interface->ndpRouterAdvContext;
423 
424  //A host must silently discard any received Router Solicitation
425  if(context == NULL)
426  return;
427 
428  //Get current time
429  time = osGetSystemTime();
430 
431  //Retrieve the length of the message
432  length = netBufferGetLength(buffer) - offset;
433 
434  //Check the length of the Router Solicitation message
435  if(length < sizeof(NdpRouterSolMessage))
436  return;
437 
438  //Point to the beginning of the message
439  message = netBufferAt(buffer, offset);
440  //Sanity check
441  if(message == NULL)
442  return;
443 
444  //Debug message
445  TRACE_INFO("Router Solicitation message received (%" PRIuSIZE " bytes)...\r\n",
446  length);
447 
448  //Dump message contents for debugging purpose
449  ndpDumpRouterSolMessage(message);
450 
451  //The IPv6 Hop Limit field must have a value of 255 to ensure that the
452  //packet has not been forwarded by a router
453  if(hopLimit != NDP_HOP_LIMIT)
454  return;
455 
456  //ICMPv6 Code must be 0
457  if(message->code)
458  return;
459 
460  //Calculate the length of the Options field
461  length -= sizeof(NdpRouterSolMessage);
462 
463  //Parse Options field
464  error = ndpCheckOptions(message->options, length);
465  //All included options must have a length that is greater than zero
466  if(error)
467  return;
468 
469  //Search for the Source Link-Layer Address option
470  option = ndpGetOption(message->options,
471  length, NDP_OPT_SOURCE_LINK_LAYER_ADDR);
472 
473  //Source Link-Layer Address option found?
474  if(option != NULL && option->length == 1)
475  {
476  //Debug message
477  TRACE_DEBUG(" Source Link-Layer Address = %s\r\n",
478  macAddrToString(&option->linkLayerAddr, NULL));
479 
480  //The Source Link-Layer Address option must not be included when the
481  //source IP address is the unspecified address
482  if(ipv6CompAddr(&pseudoHeader->srcAddr, &IPV6_UNSPECIFIED_ADDR))
483  return;
484 
485  //Search the Neighbor Cache for the source address of the solicitation
486  entry = ndpFindNeighborCacheEntry(interface, &pseudoHeader->srcAddr);
487 
488  //No matching entry has been found?
489  if(!entry)
490  {
491  //Create an entry
492  entry = ndpCreateNeighborCacheEntry(interface);
493 
494  //Neighbor Cache entry successfully created?
495  if(entry)
496  {
497  //Record the IPv6 and the corresponding MAC address
498  entry->ipAddr = pseudoHeader->srcAddr;
499  entry->macAddr = option->linkLayerAddr;
500  //The IsRouter flag must be set to FALSE
501  entry->isRouter = FALSE;
502  //Save current time
503  entry->timestamp = time;
504  //Enter the STALE state
505  entry->state = NDP_STATE_STALE;
506  }
507  }
508  else
509  {
510  //If a Neighbor Cache entry for the solicitation's sender exists the
511  //entry's IsRouter flag must be set to FALSE
512  entry->isRouter = FALSE;
513 
514  //INCOMPLETE state?
515  if(entry->state == NDP_STATE_INCOMPLETE)
516  {
517  //Record link-layer address
518  entry->macAddr = option->linkLayerAddr;
519  //Send all the packets that are pending for transmission
520  n = ndpSendQueuedPackets(interface, entry);
521  //Save current time
522  entry->timestamp = time;
523 
524  //Check whether any packets have been sent
525  if(n > 0)
526  {
527  //Start delay timer
528  entry->timeout = NDP_DELAY_FIRST_PROBE_TIME;
529  //Switch to the DELAY state
530  entry->state = NDP_STATE_DELAY;
531  }
532  else
533  {
534  //Enter the STALE state
535  entry->state = NDP_STATE_STALE;
536  }
537  }
538  //REACHABLE, STALE, DELAY or PROBE state?
539  else
540  {
541  //Different link-layer address than cached?
542  if(!macCompAddr(&entry->macAddr, &option->linkLayerAddr))
543  {
544  //Update link-layer address
545  entry->macAddr = option->linkLayerAddr;
546  //Save current time
547  entry->timestamp = time;
548  //Enter the STALE state
549  entry->state = NDP_STATE_STALE;
550  }
551  }
552  }
553  }
554 
555  //Upon receipt of a Router Solicitation, compute a random delay within the
556  //range 0 through MAX_RA_DELAY_TIME
557  delay = netGetRandRange(0, NDP_MAX_RA_DELAY_TIME);
558 
559  //If the computed value corresponds to a time later than the time the next
560  //multicast Router Advertisement is scheduled to be sent, ignore the random
561  //delay and send the advertisement at the already-scheduled time
562  if(timeCompare(time + delay, context->timestamp + context->timeout) > 0)
563  return;
564 
565  //Check whether the router sent a multicast Router Advertisement (solicited
566  //or unsolicited) within the last MIN_DELAY_BETWEEN_RAS seconds
567  if(timeCompare(time, context->timestamp + NDP_MIN_DELAY_BETWEEN_RAS) < 0)
568  {
569  //Schedule the advertisement to be sent at a time corresponding to
570  //MIN_DELAY_BETWEEN_RAS plus the random value after the previous
571  //advertisement was sent. This ensures that the multicast Router
572  //Advertisements are rate limited
573  context->timeout = NDP_MIN_DELAY_BETWEEN_RAS + delay;
574  }
575  else
576  {
577  //Schedule the sending of a Router Advertisement at the time given by the
578  //random value
579  context->timeout = time + delay - context->timestamp;
580  }
581 }

591 error_t ndpSendRouterAdv(NdpRouterAdvContext *context, uint16_t routerLifetime)
592 {
593  error_t error;
594  uint_t i;
595  uint32_t n;
596  size_t offset;
597  size_t length;
598  NetBuffer *buffer;
599  NetInterface *interface;
600  NdpRouterAdvMessage *message;
601  NdpRouterAdvSettings *settings;
602  Ipv6PseudoHeader pseudoHeader;
603 #if (ETH_SUPPORT == ENABLED)
604  NetInterface *logicalInterface;
605 #endif
606 
607  //Point to the underlying network interface
608  interface = context->settings.interface;
609  //Point to the router configuration variables
610  settings = &context->settings;
611 
612  //The destination address is typically the all-nodes multicast address
613  pseudoHeader.destAddr = IPV6_LINK_LOCAL_ALL_NODES_ADDR;
614 
615  //Routers must use their link-local address as the source for Router
616  //Advertisement messages so that hosts can uniquely identify routers
617  error = ipv6SelectSourceAddr(&interface, &pseudoHeader.destAddr,
618  &pseudoHeader.srcAddr);
619 
620  //No link-local address assigned to the interface?
621  if(error)
622  return error;
623 
624  //Compute the maximum size of the Router Advertisement message
625  length = sizeof(NdpRouterAdvMessage) +
626  sizeof(NdpLinkLayerAddrOption) + sizeof(NdpMtuOption) +
627  settings->prefixListLength * sizeof(NdpPrefixInfoOption) +
628  settings->routeListLength * sizeof(NdpRouteInfoOption) +
629  settings->contextListLength * sizeof(NdpContextOption);
630 
631  //Sanity check
632  if((length + sizeof(Ipv6Header)) > IPV6_DEFAULT_MTU)
633  return ERROR_FAILURE;
634 
635  //Allocate a memory buffer to hold the Router Advertisement message
636  buffer = ipAllocBuffer(length, &offset);
637  //Failed to allocate memory?
638  if(buffer == NULL)
639  return ERROR_OUT_OF_MEMORY;
640 
641  //Point to the beginning of the message
642  message = netBufferAt(buffer, offset);
643 
644  //Format Router Advertisement message
645  message->type = ICMPV6_TYPE_ROUTER_ADV;
646  message->code = 0;
647  message->checksum = 0;
648  message->curHopLimit = settings->curHopLimit;
649  message->m = settings->managedFlag;
650  message->o = settings->otherConfigFlag;
651  message->h = settings->homeAgentFlag;
652  message->prf = settings->preference;
653  message->p = settings->proxyFlag;
654  message->reserved = 0;
655  message->routerLifetime = htons(routerLifetime);
656  message->reachableTime = htonl(settings->reachableTime);
657  message->retransTimer = htonl(settings->retransTimer);
658 
659  //If the Router Lifetime is zero, the preference value must be set to
660  //zero by the sender
661  if(routerLifetime == 0)
662  message->prf = NDP_ROUTER_SEL_PREFERENCE_MEDIUM;
663 
664  //Length of the message, excluding any option
665  length = sizeof(NdpRouterAdvMessage);
666 
667 #if (ETH_SUPPORT == ENABLED)
668  //Point to the logical interface
669  logicalInterface = nicGetLogicalInterface(interface);
670 
671  //Check whether a MAC address has been assigned to the interface
672  if(!macCompAddr(&logicalInterface->macAddr, &MAC_UNSPECIFIED_ADDR))
673  {
674  //Add Source Link-Layer Address option
675  ndpAddOption(message, &length, NDP_OPT_SOURCE_LINK_LAYER_ADDR,
676  &logicalInterface->macAddr, sizeof(MacAddr));
677  }
678 #endif
679 
680  //A value of zero indicates that no MTU option is sent
681  if(settings->linkMtu > 0)
682  {
683  NdpMtuOption mtuOption;
684 
685  //The MTU option specifies the recommended MTU for the link
686  mtuOption.reserved = 0;
687  mtuOption.mtu = htonl(settings->linkMtu);
688 
689  //Add MTU option
690  ndpAddOption(message, &length, NDP_OPT_MTU,
691  (uint8_t *) &mtuOption + sizeof(NdpOption),
692  sizeof(NdpMtuOption) - sizeof(NdpOption));
693  }
694 
695  //Loop through the list of IPv6 prefixes
696  for(i = 0; i < settings->prefixListLength; i++)
697  {
698  NdpPrefixInfoOption prefixInfoOption;
699 
700  //The Prefix Information option provide hosts with on-link prefixes and
701  //prefixes for Address Autoconfiguration
702  prefixInfoOption.prefixLength = settings->prefixList[i].length;
703  prefixInfoOption.l = settings->prefixList[i].onLinkFlag;
704  prefixInfoOption.a = settings->prefixList[i].autonomousFlag;
705  prefixInfoOption.r = 0;
706  prefixInfoOption.reserved1 = 0;
707  prefixInfoOption.validLifetime = htonl(settings->prefixList[i].validLifetime);
708  prefixInfoOption.preferredLifetime = htonl(settings->prefixList[i].preferredLifetime);
709  prefixInfoOption.reserved2 = 0;
710  prefixInfoOption.prefix = settings->prefixList[i].prefix;
711 
712  //Add Prefix Information option (PIO)
713  ndpAddOption(message, &length, NDP_OPT_PREFIX_INFORMATION,
714  (uint8_t *) &prefixInfoOption + sizeof(NdpOption),
715  sizeof(NdpPrefixInfoOption) - sizeof(NdpOption));
716  }
717 
718  //Loop through the list of routes
719  for(i = 0; i < settings->routeListLength; i++)
720  {
721  NdpRouteInfoOption routeInfoOption;
722 
723  //The Route Information option specifies prefixes that are reachable via
724  //the router
725  routeInfoOption.prefixLength = settings->routeList[i].length;
726  routeInfoOption.reserved1 = 0;
727  routeInfoOption.prf = settings->routeList[i].preference;
728  routeInfoOption.reserved2 = 0;
729  routeInfoOption.routeLifetime = htonl(settings->routeList[i].routeLifetime);
730  routeInfoOption.prefix = settings->routeList[i].prefix;
731 
732  //Add Route Information option (RIO)
733  ndpAddOption(message, &length, NDP_OPT_ROUTE_INFORMATION,
734  (uint8_t *) &routeInfoOption + sizeof(NdpOption),
735  sizeof(NdpRouteInfoOption) - sizeof(NdpOption));
736  }
737 
738  //Loop through the list of 6LoWPAN compression contexts
739  for(i = 0; i < settings->contextListLength; i++)
740  {
741  NdpContextOption contextOption;
742 
743  //The 6LoWPAN Context option (6CO) carries prefix information for LoWPAN
744  //header compression
745  contextOption.contextLength = settings->contextList[i].length;
746  contextOption.reserved1 = 0;
747  contextOption.c = settings->contextList[i].compression;
748  contextOption.cid = settings->contextList[i].cid;
749  contextOption.reserved2 = 0;
750  contextOption.validLifetime = htons(settings->contextList[i].validLifetime);
751  contextOption.contextPrefix = settings->contextList[i].prefix;
752 
753  //Calculate the length of the option in bytes
754  n = sizeof(NdpContextOption) - sizeof(Ipv6Addr) + (contextOption.contextLength / 8);
755 
756  //Add 6LoWPAN Context option (6CO)
757  ndpAddOption(message, &length, NDP_OPT_6LOWPAN_CONTEXT,
758  (uint8_t *) &contextOption + sizeof(NdpOption), n - sizeof(NdpOption));
759  }
760 
761  //Adjust the length of the multi-part buffer
762  netBufferSetLength(buffer, offset + length);
763 
764  //Format IPv6 pseudo header
765  pseudoHeader.length = htonl(length);
766  pseudoHeader.reserved = 0;
767  pseudoHeader.nextHeader = IPV6_ICMPV6_HEADER;
768 
769  //Calculate ICMPv6 header checksum
770  message->checksum = ipCalcUpperLayerChecksumEx(&pseudoHeader,
771  sizeof(Ipv6PseudoHeader), buffer, offset, length);
772 
773  //Total number of ICMP messages which this entity attempted to send
774  IP_MIB_INC_COUNTER32(icmpv6Stats.icmpStatsOutMsgs, 1);
775  //Increment per-message type ICMP counter
776  IP_MIB_INC_COUNTER32(icmpv6MsgStatsTable.icmpMsgStatsOutPkts[ICMPV6_TYPE_ROUTER_ADV], 1);
777 
778  //Debug message
779  TRACE_INFO("Sending Router Advertisement message (%" PRIuSIZE " bytes)...\r\n",
780  length);
781 
782  //Dump message contents for debugging purpose
