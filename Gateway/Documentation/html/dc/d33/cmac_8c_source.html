184 {
185  size_t n;
186 
187  //Process the incoming data
188  while(dataLen > 0)

230 {
231  //Check the length of the MAC
232  if(macLen < 1 || macLen > context->cipher->blockSize)
233  return ERROR_INVALID_PARAMETER;
234 
235  //Check whether the last block Mn is complete
236  if(context->bufferLength >= context->cipher->blockSize)
237  {
238  //The final block M(n) is XOR-ed with the first subkey K1
239  cmacXorBlock(context->buffer, context->buffer, context->k1,
240  context->cipher->blockSize);
241  }
242  else

244  //Append padding string
245  context->buffer[context->bufferLength++] = 0x80;
246

248  while(context->bufferLength < context->cipher->blockSize)

252  cmacXorBlock(context->buffer, context->buffer, context->k2,

285 void cmacMul(uint8_t *x, const uint8_t *a, size_t n, uint8_t rb)
286 {
287  size_t i;
288  uint8_t c;
289 
290  //Save the value of the most significant bit
291  c = a[0] >> 7;
292 
293  //The multiplication of a polynomial by x in GF(2^128) corresponds to a
294  //shift of indices
295  for(i = 0; i < (n - 1); i++)
296  {
297  x[i] = (a[i] << 1) | (a[i + 1] >> 7);
298  }
299 
300  //Shift the last byte of the block to the left
301  x[i] = a[i] << 1;
302 
303  //If the highest term of the result is equal to one, then perform reduction
304  x[i] ^= rb & ~(c - 1);
305 }
