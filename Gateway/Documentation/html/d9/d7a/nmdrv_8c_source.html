231 static sint8 nm_get_firmware_info(tstrM2mRev* M2mRev)
232 {
233  uint16 curr_drv_ver, min_req_drv_ver;
234  uint32 reg = 0;
235  sint8 ret = M2M_SUCCESS;
236 
237  ret = nm_read_reg_with_ret(NMI_REV_REG, &reg);
238 
239  M2mRev->u8DriverMajor = (uint8)(reg >> 24)&0xff;
240  M2mRev->u8DriverMinor = (uint8)(reg >> 20)&0x0f;
241  M2mRev->u8DriverPatch = (uint8)(reg >> 16)&0x0f;
242  M2mRev->u8FirmwareMajor = (uint8)(reg >> 8)&0xff;
243  M2mRev->u8FirmwareMinor = (uint8)(reg >> 4)&0x0f;
244  M2mRev->u8FirmwarePatch = (uint8)(reg)&0x0f;
245  M2mRev->u32Chipid = nmi_get_chipid();
246 
247  curr_drv_ver = MAKE_VERSION(M2M_DRIVER_VERSION_MAJOR_NO, M2M_DRIVER_VERSION_MINOR_NO, M2M_DRIVER_VERSION_PATCH_NO);
248  min_req_drv_ver = MAKE_VERSION(M2mRev->u8DriverMajor, M2mRev->u8DriverMinor,M2mRev->u8DriverPatch);
249  if((curr_drv_ver < min_req_drv_ver) || M2mRev->u8DriverMajor == 18 ) {
250  ret = M2M_ERR_FW_VER_MISMATCH;
251  }
252  return ret;
253 }
