79 error_t mldStartListening(NetInterface *interface, Ipv6FilterEntry *entry)
80 {
81  //The link-scope all-nodes address (FF02::1) is handled as a special
82  //case. The host starts in Idle Listener state for that address on
83  //every interface and never transitions to another state
84  if(ipv6CompAddr(&entry->addr, &IPV6_LINK_LOCAL_ALL_NODES_ADDR))
85  {
86  //Clear flag
87  entry->flag = FALSE;
88  //Enter the Idle Listener state
89  entry->state = MLD_STATE_IDLE_LISTENER;
90  }
91  else
92  {
93  //Link is up?
94  if(interface->linkState)
95  {
96  //Send a Multicast Listener Report message for the group on the interface
97  mldSendListenerReport(interface, &entry->addr);
98 
99  //Set flag
100  entry->flag = TRUE;
101  //Start timer
102  entry->timer = osGetSystemTime() + MLD_UNSOLICITED_REPORT_INTERVAL;
103  //Enter the Delaying Listener state
104  entry->state = MLD_STATE_DELAYING_LISTENER;
105  }
106  //Link is down?
107  else
108  {
109  //Clear flag
110  entry->flag = FALSE;
111  //Enter the Idle Listener state
112  entry->state = MLD_STATE_IDLE_LISTENER;
113  }
114  }

268 void mldProcessListenerQuery(NetInterface *interface, Ipv6PseudoHeader *pseudoHeader,
269  const NetBuffer *buffer, size_t offset, uint8_t hopLimit)
270 {
271  uint_t i;
272  size_t length;
273  systime_t time;
274  systime_t maxRespDelay;
275  MldMessage *message;
276  Ipv6FilterEntry *entry;
277 
278  //Retrieve the length of the MLD message
279  length = netBufferGetLength(buffer) - offset;
280 
281  //The message must be at least 24 octets long
282  if(length < sizeof(MldMessage))
283  return;
284 
285  //Point to the beginning of the MLD message
286  message = netBufferAt(buffer, offset);
287  //Sanity check
288  if(message == NULL)
289  return;
290 
291  //Debug message
292  TRACE_INFO("MLD message received (%" PRIuSIZE " bytes)...\r\n", length);
293  //Dump message contents for debugging purpose
294  mldDumpMessage(message);
295 
296  //Make sure the source address of the message is a valid link-local address
297  if(!ipv6IsLinkLocalUnicastAddr(&pseudoHeader->srcAddr))
298  return;
299 
300  //Check the Hop Limit field
301  if(hopLimit != MLD_HOP_LIMIT)
302  return;
303 
304  //Get current time
305  time = osGetSystemTime();
306 
307  //The Max Resp Delay field specifies the maximum time allowed
308  //before sending a responding report
309  maxRespDelay = message->maxRespDelay * 10;
310 
311  //Go through the multicast filter table
312  for(i = 0; i < IPV6_MULTICAST_FILTER_SIZE; i++)
313  {
314  //Point to the current entry
315  entry = &interface->ipv6Context.multicastFilter[i];
316 
317  //Valid entry?
318  if(entry->refCount > 0)
319  {
320  //The link-scope all-nodes address (FF02::1) is handled as a special
321  //case. The host starts in Idle Listener state for that address on
322  //every interface and never transitions to another state
323  if(!ipv6CompAddr(&entry->addr, &IPV6_LINK_LOCAL_ALL_NODES_ADDR))
324  {
325  //A General Query is used to learn which multicast addresses have listeners
326  //on an attached link. A Multicast-Address-Specific Query is used to learn
327  //if a particular multicast address has any listeners on an attached link
328  if(ipv6CompAddr(&message->multicastAddr, &IPV6_UNSPECIFIED_ADDR) ||
329  ipv6CompAddr(&message->multicastAddr, &entry->addr))
330  {
331  //Delaying Listener state?
332  if(entry->state == MLD_STATE_DELAYING_LISTENER)
333  {
334  //The timer has not yet expired?
335  if(timeCompare(time, entry->timer) < 0)
336  {
337  //If a timer for the address is already running, it is reset to
338  //the new random value only if the requested Max Response Delay
339  //is less than the remaining value of the running timer
340  if(maxRespDelay < (entry->timer - time))
341  {
342  //Restart delay timer
343  entry->timer = time + mldRand(maxRespDelay);
344  }
345  }
346  }
347  //Idle Listener state?
348  else if(entry->state == MLD_STATE_IDLE_LISTENER)
349  {
350  //Switch to the Delaying Listener state
351  entry->state = MLD_STATE_DELAYING_LISTENER;
352  //Delay the response by a random amount of time
353  entry->timer = time + mldRand(maxRespDelay);
354  }
355  }
356  }
357  }
358  }

438 error_t mldSendListenerReport(NetInterface *interface, Ipv6Addr *ipAddr)
439 {
440  error_t error;
441  size_t offset;
442  MldMessage *message;
443  NetBuffer *buffer;
444  Ipv6PseudoHeader pseudoHeader;
445 
446  //Make sure the specified address is a valid multicast address
447  if(!ipv6IsMulticastAddr(ipAddr))
448  return ERROR_INVALID_ADDRESS;
449 
450  //The link-scope all-nodes address (FF02::1) is handled as a special
451  //case. The host never sends a report for that address
452  if(ipv6CompAddr(ipAddr, &IPV6_LINK_LOCAL_ALL_NODES_ADDR))
453  return ERROR_INVALID_ADDRESS;
454 
455  //Allocate a memory buffer to hold a MLD message
456  buffer = ipAllocBuffer(sizeof(MldMessage), &offset);
457  //Failed to allocate memory?
458  if(buffer == NULL)
459  return ERROR_OUT_OF_MEMORY;
460 
461  //Point to the beginning of the MLD message
462  message = netBufferAt(buffer, offset);
463 
464  //Format the Multicast Listener Report message
465  message->type = ICMPV6_TYPE_MULTICAST_LISTENER_REPORT_V1;
466  message->code = 0;
467  message->checksum = 0;
468  message->maxRespDelay = 0;
469  message->reserved = 0;
470  message->multicastAddr = *ipAddr;
471 
472  //Format IPv6 pseudo header
473  pseudoHeader.srcAddr = interface->ipv6Context.addrList[0].addr;
474  pseudoHeader.destAddr = *ipAddr;
475  pseudoHeader.length = HTONS(sizeof(MldMessage));
476  pseudoHeader.reserved = 0;
477  pseudoHeader.nextHeader = IPV6_ICMPV6_HEADER;
478 
479  //Message checksum calculation
480  message->checksum = ipCalcUpperLayerChecksumEx(&pseudoHeader,
481  sizeof(Ipv6PseudoHeader), buffer, offset, sizeof(MldMessage));
482 
483  //Total number of ICMP messages which this entity attempted to send
484  IP_MIB_INC_COUNTER32(icmpv6Stats.icmpStatsOutMsgs, 1);
485 
486  //Increment per-message type ICMP counter
487  IP_MIB_INC_COUNTER32(icmpv6MsgStatsTable.icmpMsgStatsOutPkts[
488  ICMPV6_TYPE_MULTICAST_LISTENER_REPORT_V1], 1);
489 
490  //Debug message
491  TRACE_INFO("Sending MLD message (%" PRIuSIZE " bytes)...\r\n", sizeof(MldMessage));
492  //Dump message contents for debugging purpose
493  mldDumpMessage(message);
494 
495  //The Multicast Listener Report message is sent to the multicast address being reported
496  error = ipv6SendDatagram(interface, &pseudoHeader, buffer, offset, MLD_HOP_LIMIT);
497 
498  //Free previously allocated memory
499  netBufferFree(buffer);
500  //Return status code
501  return error;
502 }
