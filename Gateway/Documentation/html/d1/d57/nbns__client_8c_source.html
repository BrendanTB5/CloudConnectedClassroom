204  buffer = udpAllocBuffer(DNS_MESSAGE_MAX_SIZE, &offset);
205  //Failed to allocate buffer?
206  if(buffer == NULL)
207  return ERROR_OUT_OF_MEMORY;
208

210  message = netBufferAt(buffer, offset);
211 
212  //Format NBNS query message

214  message->qr = 0;
215  message->opcode = DNS_OPCODE_QUERY;
216  message->aa = 0;
217  message->tc = 0;
218  message->rd = 0;
219  message->ra = 0;

221  message->b = 1;
222  message->rcode = DNS_RCODE_NO_ERROR;
223 
224  //The NBNS query contains one question
225  message->qdcount = HTONS(1);
226  message->ancount = 0;
227  message->nscount = 0;
228  message->arcount = 0;
229 
230  //Length of the NBNS query message
231  length = sizeof(DnsHeader);
232 
233  //Encode the NetBIOS name
234  length += nbnsEncodeName(entry->name, message->questions);
235 
236  //Point to the corresponding question structure
237  dnsQuestion = DNS_GET_QUESTION(message, length);
238  //Fill in question structure
239  dnsQuestion->qtype = HTONS(DNS_RR_TYPE_NB);
240  dnsQuestion->qclass = HTONS(DNS_RR_CLASS_IN);
241 
242  //Update the length of the NBNS query message
243  length += sizeof(DnsQuestion);
244 
245  //Adjust the length of the multi-part buffer
246  netBufferSetLength(buffer, offset + length);
247 
248  //Debug message
249  TRACE_INFO("Sending NBNS message (%" PRIuSIZE " bytes)...\r\n", length);
250  //Dump message
251  dnsDumpMessage((DnsHeader *) message, length);
252 
253  //The destination address is the broadcast address
254  destIpAddr.length = sizeof(Ipv4Addr);
255  ipv4GetBroadcastAddr(entry->interface, &destIpAddr.ipv4Addr);
256 
257  //A request packet is always sent to the well known port 137
258  error = udpSendDatagramEx(entry->interface, NULL, NBNS_PORT,
259  &destIpAddr, NBNS_PORT, buffer, offset, IPV4_DEFAULT_TTL);
260 
261  //Free previously allocated memory
262  netBufferFree(buffer);
263  //Return status code

283  DnsResourceRecord *record;

285 
286  //The NBNS response shall contain one answer
287  if(ntohs(message->qdcount) != 0 && ntohs(message->ancount) != 1)
288  return;
289 
290  //Parse NetBIOS name
291  pos = nbnsParseName(message, length, sizeof(DnsHeader), NULL);
292  //Invalid name?
293  if(!pos)
294  return;
295 
296  //Point to the associated resource record
297  record = DNS_GET_RESOURCE_RECORD(message, pos);
298  //Point to the resource data
299  pos += sizeof(DnsResourceRecord);
300 
301  //Make sure the resource record is valid
302  if(pos > length)
303  return;
304  if((pos + ntohs(record->rdlength)) > length)
305  return;
306 
307  //Check the class and the type of the resource record
308  if(ntohs(record->rclass) != DNS_RR_CLASS_IN)
309  return;
310  if(ntohs(record->rtype) != DNS_RR_TYPE_NB)
311  return;
312 
313  //Verify the length of the data field
314  if(ntohs(record->rdlength) < sizeof(NbnsAddrEntry))
315  return;
316 
317  //Loop through DNS cache entries
318  for(i = 0; i < DNS_CACHE_SIZE; i++)
319  {
320  //Point to the current entry
321  entry = &dnsCache[i];
322 
323  //NBNS name resolution in progress?
324  if(entry->state == DNS_STATE_IN_PROGRESS &&
325  entry->protocol == HOST_NAME_RESOLVER_NBNS &&
326  entry->type == HOST_TYPE_IPV4)
327  {
328  //Compare identifiers
329  if(entry->id == ntohs(message->id))
330  {
331  //Compare NetBIOS names
332  if(nbnsCompareName(message, length, sizeof(DnsHeader), entry->name))
333  {

339

342  //Save TTL value

344  //Limit the lifetime of the NBNS cache entries
345  entry->timeout = MIN(entry->timeout, NBNS_MAX_LIFETIME);
346
