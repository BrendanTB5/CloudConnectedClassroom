333  gstrHif.u16Length = M2M_HIF_HDR_OFFSET;
334  if(pu8DataBuf != NULL)

336  gstrHif.u16Length += u16DataOffset + u16DataSize;

361  if(pu8CtrlBuf != NULL)
362  {

366  }

484 static sint8 hif_isr(void)
485 {
486  sint8 ret = M2M_ERR_BUS_FAIL;
487  volatile uint32 reg,int_stat;
488  volatile tstrHifHdr* pstrHif;
489 
490  /*read int status*/
491  ret = nm_read_reg_with_ret(WILC_INT_STATUS_REG, (uint32*)&int_stat);
492  int_stat >>= IRG_FLAGS_OFFSET;
493  /*clear the read status*/
494  ret |= nm_write_reg(WILC_INT_CLEAR_REG,int_stat);
495  if(M2M_SUCCESS == ret)
496  {
497 #ifdef INT_BASED_TX
498  if(int_stat & 0x2) /* Packet sent. Host can now send more packets*/
499  {
500  if( (int_stat&1) == 0)
501  nm_bsp_interrupt_ctrl(1);
502 
503  if(pfWifiCb)
504  pfWifiCb(M2M_WIFI_RESP_PACKET_SENT,0,0);
505  }
506 #endif
507  if(int_stat & 0x1) /* New interrupt has been received */
508  {
509  uint16 size;
510  /*read packet size*/
511  ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_0, (uint32*)&reg);
512  nm_bsp_interrupt_ctrl(0);
513  gu8HifSizeDone = 1;
514  size = (uint16)((reg >> 2) & 0xfff);
515  if (size > 0) {
516  uint32 address = 0;
520  ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_1, &address);
521  if((M2M_SUCCESS != ret) || (size > strHifInitParam.u32RcvBuffSize))
522  {
523  M2M_ERR("(hif) WIFI_HOST_RCV_CTRL_1 bus fail or buffer is too small. packet Discarded\r\n");
524  M2M_ERR("ret: %d, size: %d, buffersize: %lu\r\n", ret, size, strHifInitParam.u32RcvBuffSize);
525  hif_set_rx_done();
526  nm_bsp_interrupt_ctrl(1);
527  goto ERR1;
528  }
529  ret = nm_read_block(address, strHifInitParam.pu8RcvBuff, size);
530  if(M2M_SUCCESS != ret)
531  {
532  M2M_ERR("(hif) address bus fail\r\n");
533  hif_set_rx_done();
534  nm_bsp_interrupt_ctrl(1);
535  goto ERR1;
536  }
537  pstrHif = (tstrHifHdr*)strHifInitParam.pu8RcvBuff;
538  pstrHif->u16Length = NM_BSP_B_L_16(pstrHif->u16Length);
539  if(pstrHif->u16Length != size)
540  {
541  if((size - pstrHif->u16Length) > 4)
542  {
543  M2M_ERR("(hif) Corrupted packet Address: %08lx, xSize = %u <L = %u, G = %u, OP = %02X>\r\n",
544  address, size, pstrHif->u16Length, pstrHif->u8Gid, pstrHif->u8Opcode);
545  hif_set_rx_done();
546  nm_bsp_interrupt_ctrl(1);
547  ret = M2M_ERR_BUS_FAIL;
548  goto ERR1;
549  }
550  }
551 
552  if(M2M_REQ_GRP_WIFI == pstrHif->u8Gid)
553  {
554  if(pfWifiCb)
555  pfWifiCb(pstrHif->u8Opcode,pstrHif->u16Length - M2M_HIF_HDR_OFFSET, strHifInitParam.pu8RcvBuff + M2M_HIF_HDR_OFFSET);
556 
557  }
558  else
559  {
560  M2M_ERR("(hif) invalid group ID\r\n");
561  hif_set_rx_done();
562  nm_bsp_interrupt_ctrl(1);
563  ret = M2M_ERR_BUS_FAIL;
564  goto ERR1;
565  }
566  ret = hif_set_rx_done();
567  }
568  else
569  {
570  M2M_ERR("(hif) Wrong Size\r\n");
571  ret = M2M_ERR_RCV;
572  goto ERR1;
573  }
574  }
575  if (int_stat > 3 || int_stat == 0)
576  {
577 #ifndef WIN32
578  M2M_ERR("(hif) False interrupt %lx",int_stat);
579 #ifdef NM_LEVEL_INTERRUPT
580  nm_bsp_interrupt_ctrl(1);
581 
582 #endif
583 #endif
584  }
585  }
586  else
587  {
588  M2M_ERR("(hif) Fail to Read interrupt reg\r\n");
589  goto ERR1;
590  }
591 ERR1:
592  return ret;
593 }
