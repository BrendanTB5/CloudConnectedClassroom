83 void SHA1ProcessBlock(uint32* pu32Sha1State, const uint8* pu8MessageBlock)
84 {
85  /* Constants defined in FIPS-180-2, section 4.2.1 */
86  const uint32 K[4] = {
87  0x5A827999,
88  0x6ED9EBA1,
89  0x8F1BBCDC,
90  0xCA62C1D6
91  };
92  int t; /* Loop counter */
93  uint32 W[80]; /* Word sequence */
94  uint32 A, B, C, D, E; /* Word buffers */
95 
96  /* Initialize the first 16 words in the array W. */
97  for (t = 0; t < 16; t++)
98  W[t] = GETU32(pu8MessageBlock, (t * 4));
99 
100  for (t = 16; t < 80; t++)
101  W[t] = SHA_ROTL((W[t-3] ^ W[t-8] ^ W[t-14] ^ W[t-16]), 1);
102 
103  A = pu32Sha1State[0];
104  B = pu32Sha1State[1];
105  C = pu32Sha1State[2];
106  D = pu32Sha1State[3];
107  E = pu32Sha1State[4];
108 
109  for (t = 0; t < 20; t++) {
110  W[t] += SHA1_CH(B, C, D);
111  SHA1_INERMEDIATE(A, B, C, D, E, W[t], K[0])
112  }
113 
114  for (t = 20; t < 40; t++) {
115  W[t] += SHA_PARITY(B, C, D);
116  SHA1_INERMEDIATE(A, B, C, D, E, W[t], K[1])
117  }
118 
119  for (t = 40; t < 60; t++) {
120  W[t] += SHA1_MAJ(B, C, D);
121  SHA1_INERMEDIATE(A, B, C, D, E, W[t], K[2])
122  }
123 
124  for (t = 60; t < 80; t++) {
125  W[t] += SHA_PARITY(B, C, D);
126  SHA1_INERMEDIATE(A, B, C, D, E, W[t], K[3])
127  }
128 
129  pu32Sha1State[0] += A;
130  pu32Sha1State[1] += B;
131  pu32Sha1State[2] += C;
132  pu32Sha1State[3] += D;

166  u32DataLength = 0;
167  }
168  }
169

171  u32NBlocks = u32DataLength / pstrHashCxt->u8BlockSize;
172  u32ResidualBytes = u32DataLength % pstrHashCxt->u8BlockSize;
173  for(u32BlockCount = 0 ; u32BlockCount < u32NBlocks ; u32BlockCount ++)
174  {
175  pstrHashCxt->fpHash(pstrHashCxt->au32HashState, pu8Data);

338 uint8 pbkdf2_sha1_f(uint8 *key, uint8 key_length, uint8 *data, uint32 data_len,
339  uint32 iterations, uint32 count, uint8 *digest)
340 {
341  uint8 tmp1[36] = {0};
342  uint8 tmp2[A_SHA_DIGEST_LEN] = {0};
343  uint16 i, j;
344 
345  for (i = 0; i < key_length ; i++)
346  {
347  if (key[i] < 32)
348  {
349  return 0;
350  }
351  else if (key[i] > 126)
352  {
353  return 0;
354  }
355  }
356 
357 
358  /* U1 = PRF(P, S || int(i)) */
359  m2m_memcpy(tmp1, data, data_len);
360 
361  tmp1[data_len] = (uint8)((count>>24) & 0xFF);
362  tmp1[data_len+1] = (uint8)((count>>16) & 0xFF);
363  tmp1[data_len+2] = (uint8)((count>>8) & 0xFF);
364  tmp1[data_len+3] = (uint8)(count & 0xFF);
365 
366  DRV_hmac_sha1(key, key_length, tmp1, (data_len + 4), tmp2);
367 
368 
369  /* digest = U1 */
370  m2m_memcpy(digest, tmp2, A_SHA_DIGEST_LEN);
371 
372  for (i = 1; i < iterations; i++)
373  {
374  /* Un = PRF(P, Un-1) */
375  DRV_hmac_sha1(key, key_length, tmp2, A_SHA_DIGEST_LEN, tmp1);
376 
377  m2m_memcpy(tmp2, tmp1, A_SHA_DIGEST_LEN);
378 
379  /* digest = digest xor Un */
380  for (j = 0; j < A_SHA_DIGEST_LEN; j++)
381  {
382  digest[j] ^= tmp1[j];
383  }
384  }
385  return 1;
386 }
