186 void lan8742EventHandler(NetInterface *interface)
187 {
188  uint16_t value;
189 
190  //Read status register to acknowledge the interrupt
191  value = lan8742ReadPhyReg(interface, LAN8742_ISR);
192 
193  //Link status change?
194  if(value & (LAN8742_IMR_AN_COMPLETE | LAN8742_IMR_LINK_DOWN))
195  {
196  //Any link failure condition is latched in the BMSR register. Reading
197  //the register twice will always return the actual link status
198  value = lan8742ReadPhyReg(interface, LAN8742_BMSR);
199  value = lan8742ReadPhyReg(interface, LAN8742_BMSR);
200 
201  //Link is up?
202  if(value & LAN8742_BMSR_LINK_STATUS)
203  {
204  //Read PHY special control/status register
205  value = lan8742ReadPhyReg(interface, LAN8742_PSCSR);
206 
207  //Check current operation mode
208  switch(value & LAN8742_PSCSR_HCDSPEED)
209  {
210  //10BASE-T half-duplex
211  case LAN8742_PSCSR_HCDSPEED_10BT_HD:
212  interface->linkSpeed = NIC_LINK_SPEED_10MBPS;
213  interface->duplexMode = NIC_HALF_DUPLEX_MODE;
214  break;
215  //10BASE-T full-duplex
216  case LAN8742_PSCSR_HCDSPEED_10BT_FD:
217  interface->linkSpeed = NIC_LINK_SPEED_10MBPS;
218  interface->duplexMode = NIC_FULL_DUPLEX_MODE;
219  break;
220  //100BASE-TX half-duplex
221  case LAN8742_PSCSR_HCDSPEED_100BTX_HD:
222  interface->linkSpeed = NIC_LINK_SPEED_100MBPS;
223  interface->duplexMode = NIC_HALF_DUPLEX_MODE;
224  break;
225  //100BASE-TX full-duplex
226  case LAN8742_PSCSR_HCDSPEED_100BTX_FD:
227  interface->linkSpeed = NIC_LINK_SPEED_100MBPS;
228  interface->duplexMode = NIC_FULL_DUPLEX_MODE;
229  break;
230  //Unknown operation mode
231  default:
232  //Debug message
233  TRACE_WARNING("Invalid operation mode!\r\n");
234  break;
235  }
236 
237  //Update link state
238  interface->linkState = TRUE;
239 
240  //Adjust MAC configuration parameters for proper operation
241  interface->nicDriver->updateMacConfig(interface);
242  }
243  else
244  {
245  //Update link state
246  interface->linkState = FALSE;
247  }
248 
249  //Process link state change event

251  }
