568  if(n < DTLS_REPLAY_WINDOW_SIZE)

580 #if (DTLS_REPLAY_DETECTION_SUPPORT == ENABLED)

584

586  n -= right;
587 
588  //Check resulting value
589  if(n < DTLS_REPLAY_WINDOW_SIZE)
590  {
591  j = (uint_t) (n / 32);
592  k = (uint_t) (n % 32);
593 
594  //First, shift words
595  if(j > 0)

598  for(i = (DTLS_REPLAY_WINDOW_SIZE - 1) / 32; i >= j; i--)

622  }
