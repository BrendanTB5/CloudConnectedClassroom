242 {

244  uint32_t t;
245  uint32_t key0;
246  uint32_t key1;
247  uint32_t key2;
248  uint32_t key3;
249 
250  //Check parameters
251  if(context == NULL || key == NULL)
252  return ERROR_INVALID_PARAMETER;
253 
254  //Invalid key length?
255  if(keyLen != 16)
256  return ERROR_INVALID_KEY_LENGTH;
257 
258  //The 128-bit input key is divided into four 32-bit blocks
259  key0 = LOAD32BE(key + 0);
260  key1 = LOAD32BE(key + 4);
261  key2 = LOAD32BE(key + 8);
262  key3 = LOAD32BE(key + 12);
263 
264  //Apply 16 rounds
265  for(i = 0; i < 16; i++)
266  {
267  t = key0 + key2 - kc[i];
268  context->ks[2 * i] = G(t);
269  t = key1 - key3 + kc[i];
270  context->ks[2 * i + 1] = G(t);
271 
272  //Odd round?
273  if((i % 2) != 0)
274  {
275  t = (key3 << 8) | (key2 >> 24);
276  key2 = (key2 << 8) | (key3 >> 24);
277  key3 = t;
278  }
279  //Even round?
280  else
281  {
282  t = (key1 >> 8) | (key0 << 24);
283  key0 = (key0 >> 8) | (key1 << 24);
284  key1 = t;
285  }
286  }

288  //No error to report
289  return NO_ERROR;
