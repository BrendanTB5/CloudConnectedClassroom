376 sint8 chip_wake(void)
377 {
378  sint8 ret;
379  volatile uint32 reg = 0, clk_status_reg = 0,trials = 0;
380 
381 #ifdef CONF_WILC_USE_1000_REV_B
382  ret = nm_read_reg_with_ret(WILC_FROM_INTERFACE_TO_WF_REG, (uint32*)&reg);
383  if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
384 
385  if(!(reg & WILC_FROM_INTERFACE_TO_WF_BIT))
386  {
387  /*USE bit 0 to indicate host wakeup*/
388  ret = nm_write_reg(WILC_FROM_INTERFACE_TO_WF_REG, reg|WILC_FROM_INTERFACE_TO_WF_BIT);
389  if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
390  }
391 #endif
392  ret = nm_read_reg_with_ret(WILC_WAKEUP_REG, (uint32*)&reg);
393  if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
394  /* Set bit 1 */
395  if(!(reg & WILC_WAKEUP_BIT))
396  {
397  ret = nm_write_reg(WILC_WAKEUP_REG, reg | WILC_WAKEUP_BIT);
398  if(ret != M2M_SUCCESS) goto _WAKE_EXIT;
399  }
400 
401  do
402  {
403  ret = nm_read_reg_with_ret(WILC_CLK_STATUS_REG, (uint32*)&clk_status_reg);
404  if(ret != M2M_SUCCESS) {
405  M2M_ERR("Bus error (5).%d %lx\r\n",ret,clk_status_reg);
406  goto _WAKE_EXIT;
407  }
408  if(clk_status_reg & WILC_CLK_STATUS_BIT) {
409  break;
410  }
411  //nm_bsp_sleep(2);
412  trials++;
413  if(trials > WAKUP_TRAILS_TIMEOUT)
414  {
415  M2M_ERR("Failed to wakup the chip\r\n");
416  ret = M2M_ERR_TIME_OUT;
417  goto _WAKE_EXIT;
418  }
419  }while(1);
420 
421  /*workaround sometimes spi fail to read clock regs after reading/writing clockless registers*/
422  nm_bus_reset();
423 
424 _WAKE_EXIT:
425  return ret;
426 }
